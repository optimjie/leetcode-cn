双指针

```java
class Solution {
    final int inf = (int)1e9;
    public int minSubArrayLen(int s, int[] nums) {
        int n = nums.length;
        int ans = inf, sum = 0;
        for (int i = 0, j = 0; i < n; i++) {
            if (i > 0) sum -= nums[i - 1];
            while (j < n && sum < s) sum += nums[j++];
            if (sum >= s) ans = Math.min(ans, j - i);
        }
        if (ans == inf) ans = 0;
        return ans;
    }
}
```
