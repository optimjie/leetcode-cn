typedef long long LL;
class Solution {
public:
    int reverse(int x) {
        if (x == INT_MIN) return 0;
        bool isNeg = false;
        if (x < 0) x = -x, isNeg = true;
        LL res = 0;
        while (x) {
            res = res * 10 + x % 10;
            x /= 10;
        }
        if (res >= INT_MIN && res <= INT_MAX) return isNeg ? 0 - res : res;
        return 0;
    } 
};
